35static inline uint32_t snrt_dma_start_1d(uint64_t dst, uint64_t src,
36 size_t size,
37 const uint32_t channel = 0) {
38#ifdef SNRT_SUPPORTS_DMA
39 uint32_t dst_lo = dst & 0xFFFFFFFF;
40 uint32_t dst_hi = dst >> 32;
41 uint32_t src_lo = src & 0xFFFFFFFF;
42 uint32_t src_hi = src >> 32;
43 uint32_t txid;
44
45 asm volatile(
46 "dmsrc %[src_lo], %[src_hi] \n"
47 "dmdst %[dst_lo], %[dst_hi] \n"
48 "dmcpyi %[txid], %[size], (%[channel] << 2) | 0b00 \n"
49 : [ txid ] "=r"(txid)
50 : [ src_lo ] "r"(src_lo), [ src_hi ] "r"(src_hi),
51 [ dst_lo ] "r"(dst_lo), [ dst_hi ] "r"(dst_hi), [ size ] "r"(size),
52 [ channel ] "i"(channel));
53
54 return txid;
55#else
56 memcpy((void *)dst, (const void *)src, size);
57 return 0;
58#endif
59}

143static inline snrt_dma_txid_t snrt_dma_start_2d(uint64_t dst, uint64_t src,
144 size_t size, size_t dst_stride,
145 size_t src_stride,
146 size_t repeat,
147 const uint32_t channel = 0) {
148#ifdef SNRT_SUPPORTS_DMA
149 uint32_t dst_lo = dst & 0xFFFFFFFF;
150 uint32_t dst_hi = dst >> 32;
151 uint32_t src_lo = src & 0xFFFFFFFF;
152 uint32_t src_hi = src >> 32;
153 uint32_t txid;
154
155 asm volatile(
156 "dmsrc %[src_lo], %[src_hi] \n"
157 "dmdst %[dst_lo], %[dst_hi] \n"
158 "dmstr %[src_stride], %[dst_stride] \n"
159 "dmrep %[repeat] \n"
160 "dmcpyi %[txid], %[size], (%[channel] << 2) | 0b10 \n"
161 : [ txid ] "=r"(txid)
162 : [ src_lo ] "r"(src_lo), [ src_hi ] "r"(src_hi),
163 [ dst_lo ] "r"(dst_lo), [ dst_hi ] "r"(dst_hi),
164 [ dst_stride ] "r"(dst_stride), [ src_stride ] "r"(src_stride),
165 [ repeat ] "r"(repeat), [ size ] "r"(size), [ channel ] "i"(channel));
166
167 return txid;
168#else
169 // TODO(colluca): we can implement this as a series of memcpy calls
170 return 0;
171#endif
172}

262#ifdef SNRT_SUPPORTS_DMA
263 uint32_t busy;
264 asm volatile(
265 "1: \n"
266 "dmstati %[busy], (%[channel] << 2) | 2 \n"

314inline void snrt_dma_memset(void *ptr, uint8_t value, uint32_t len) {
315#ifdef SNRT_SUPPORTS_DMA
316 // We set the first 64 bytes to the value, and then we use the DMA to copy
317 // these into the remaining memory region. DMA is used only if len is
318 // larger than 64 bytes, and an integer multiple of 64 bytes.
319 size_t n_1d_transfers = len / 64;
320 size_t use_dma = (len % 64) == 0 && len > 64;
321 uint8_t *p = (uint8_t *)ptr;
322
323 uint32_t nbytes = len < 64 || !use_dma ? len : 64;
324 while (nbytes--) {
325 *p++ = value;
326 }
327
328 if (use_dma) {
329 snrt_dma_start_2d(ptr, ptr, 64, 64, 0, n_1d_transfers);
330 snrt_dma_wait_all();
331 }
332#else
333 memset(ptr, (int)value, len);
334#endif
335}

435inline snrt_dma_txid_t snrt_dma_load_2d_tile(
436 void *dst, void *src, size_t tile_x1_idx, size_t tile_x0_idx,
437 size_t tile_x1_size, size_t tile_x0_size, size_t full_x0_size,
438 uint32_t prec, size_t tile_ld) {
439 size_t src_offset = 0;
440 // Advance src array in x0 and x1 dimensions, and convert to byte offset
441 src_offset += tile_x0_idx * tile_x0_size;
442 src_offset += tile_x1_idx * tile_x1_size * full_x0_size;
443 src_offset *= prec;
444 // Initiate transfer
445 return snrt_dma_start_2d((uint64_t)dst, // dst
446 (uint64_t)src + src_offset, // src
447 tile_x0_size * prec, // size
448 tile_ld, // dst_stride
449 full_x0_size * prec, // src_stride
450 tile_x1_size // repeat
451 );
452}
