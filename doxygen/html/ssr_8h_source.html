140static inline uint32_t read_ssr_cfg(const snrt_ssr_reg_t reg,
141 const snrt_ssr_dm_t dm) {
142#ifdef SNRT_SUPPORTS_SSR
143 uint32_t value;
144 asm volatile("scfgri %[value], %[dm] | %[reg]<<5\n"
145 : [ value ] "=r"(value)
146 : [ dm ] "i"(dm), [ reg ] "i"(reg));
147 return value;
148#else
149 return 0;
150#endif
151}
